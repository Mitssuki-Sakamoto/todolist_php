DROP TABLE IF EXISTS users;
DROP TABLE IF EXISTS tasks;
DROP TABLE IF EXISTS user_tasks;
CREATE TABLE users (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(20) UNIQUE NOT NULL, password VARCHAR(100) NOT NULL);
CREATE TABLE tasks (id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT, name VARCHAR(30) NOT NULL, done TINYINT NOT NULL);
CREATE TABLE user_tasks (user INT REFERENCES users(id), task INT REFERENCES tasks(id));
SHOW TABLES;
